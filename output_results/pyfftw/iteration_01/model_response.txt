```bash
# Check basic Python info and pip status
python --version
pip --version

# List primary dependency files for prioritization
ls -l requirements.txt pyproject.toml setup.py pdm.lock

# Install dependencies using pip and project-supplied files in best-effort order
pip install -e .  # Most builds respect PEP 517/518 and will use pyproject.toml and setup.py
# If not enough, also attempt requirements.txt as fallback if present
if [ -f requirements.txt ]; then pip install -r requirements.txt; fi

# Now, run pytest and observe output
pytest
```
